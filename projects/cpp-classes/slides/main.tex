\section{Inventing Classes}

\begin{frame}[fragile]{What is a Class?}
  \begin{sqt:definition}[\textbf{Definition:} Class]
    A schema/blueprint for an object, as well as a collection of functions
    (called methods) where the first argument is a pointer to an instance of that schema.
  \end{sqt:definition}

  \begin{minted}{cpp}
    struct Rectangle {
        int width_;
        int height_; 
    };

    int Rectangle_area(Rectangle *this) {
        return this->width_ * this->height_;
    }

    void Rectangle_scale(Rectangle *this, int factor) {
        this->width_ *= factor;
        this->height_ *= factor;
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Constructors}
  \begin{sqt:definition}[\textbf{Definition:} Constructor]
    A method that initializes all members of a class instance.
    The constructor should be called immediately after allocating memory for the instance.
  \end{sqt:definition}

  There can be multiple constructors with different signatures.

  \begin{minted}{cpp}
    void Rectangle_ctor_i(Rectangle *this, int side_length) {
        this->width_ = side_length;
        this->height_ = side_length;
    }

    void Rectangle_ctor_ii(Rectangle *this, int width, int height) {
        this->width_ = width;
        this->height_ = height;
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Destructors}
  \begin{sqt:definition}[\textbf{Definition:} Destructor]
    The destructor is a function that must be called on an instance when:
    \begin{itemize}
      \item A stack-allocated instance goes out of scope.
      \item A heap-allocated instance is freed.
    \end{itemize}
  \end{sqt:definition}

  There may only be one destructor per class (no overloading).

  \begin{minted}{cpp}
    void Rectangle_dtor(Rectangle *this) {
        printf("Rectangle destroyed");
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Example Class Lifecycle}
  \begin{minted}{cpp}
    {
        // alice stack allocated: must be constructed
        Rectangle alice;
        Rectangle_ctor_ii(&alice, 10, 20);
        Rectangle *bob_ptr;
        {
            // charlie stack allocated: must be constructed
            Rectangle charlie;;
            Rectangle_ctor_ii(&charlie, 5, 15);
            // bob malloced: must be constructed
            bob_ptr = (Rectangle *)malloc(sizeof(Rectangle));
            Rectangle_ctor_i(bob_ptr, 30);
            // charlie goes out of scope: must be destroyed
            Rectangle_dtor(&charlie);
        }
        // bob freed: must be destroyed
        Rectangle_dtor(bob_ptr);
        free(bob_ptr);
        // alice goes out of scope: must be destroyed
        Rectangle_dtor(&alice);
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Constructor/Destructor Semantics}
  Technically you can write constructors and destructors that do whatever you want.
  However, the convention is:
  \begin{itemize}
    \item Constructors initialize the class into a ``valid'' state
    \item Destructors free any resources held by the instance (e.g. heap memory)
  \end{itemize}

  ``Valid'' in this context means whatever the programmer wants it to mean.

\end{frame}

\begin{frame}[fragile]{Constructor Example}
  A programmer wants to implement a dynamic array of integers in C:
  \begin{minted}{cpp}
    struct dynamic_array {
        int *data_;
        size_t size_;
        size_t capacity_;
    };
  \end{minted}

  The programmer wants the object to satisfy the following properties at all times:
  \begin{itemize}
    \item \verb|size_| $\leq$ \verb|capacity_|
    \item \verb|data_| points to a heap-allocated array with length \verb|capacity_|
  \end{itemize}

  \begin{minted}{cpp}
    void dynamic_array_ctor(dynamic_array *this) {
        this->data_ = (int *)malloc(67 * sizeof(int));
        this->size_ = 0;
        this->capacity_ = 67;
    }
  \end{minted}
\end{frame}


\begin{frame}[fragile]{Destructor Example}
  Assume that the \verb|dynamic_array| instance has ``ownership'' of the memory pointed to by \verb|data_|.\\
  After instance goes out of scope or is freed, there is no way to refer to the memory pointed to by \verb|data_|.\\
  It is therefore a good idea to free that memory in the destructor.

  \begin{minted}{cpp}
    void dynamic_array_dtor(dynamic_array *this) {
        free(this->data_);
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{C++ Classes}
  C++ classes are just syntactic sugar for the C-style classes we have seen so far\\
  and ``automatically'' call constructors and destructors at the appropriate times.\\
  It also ``automatically'' prepends the \verb|this| pointer appropriately.
  \vspace{0.5cm}

  Furthermore, using \verb|new| and \verb|delete| in C++ automatically calls constructors and destructors respectively.
\end{frame}

\begin{frame}[fragile]{C++ Class Definition Example}
  \begin{minted}{cpp}
    struct Rectangle {
        int width_;
        int height_;
        Rectangle(int width, int height) { // ctor
            width_ = width;
            height_ = height;
        }
        ~Rectangle() { // dtor
            printf("Rectangle destroyed");
        }
        int area() const {
            return width_ * height_;
        }
        void scale(int factor) {
            width_ *= factor;
            height_ *= factor;
        }
    };
  \end{minted}
\end{frame}

\begin{frame}[fragile]{C++ Class Usage Example}
  \begin{minted}{cpp}
    {
        Rectangle alice{10, 20};
        Rectangle *bob_ptr;
        {
            Rectangle charlie{5, 15};
            bob_ptr = new Rectangle(30);
            // charlie goes out of scope: automatically destroyed
        }
        // delete calls destructor
        delete bob_ptr;
        // alice goes out of scope: automatically destroyed
    }
  \end{minted}
\end{frame}

\section{Ownership}

\begin{frame}[fragile]{Ownership}
  \begin{sqt:definition}[\textbf{Definition:} Ownership]
    An object/resource can be owned by 
    \begin{itemize}
      \item a scope
      \item an object
      \item shared by multiple objects
    \end{itemize}
    The owner is responsible for freeing the object/resource.
  \end{sqt:definition}

  Ownership is not a language feature, but rather a concept used to reason about a program.
  Ownership of a resource can be transferred.
\end{frame}

\section{Inheritance}
\begin{frame}[fragile]{Inheritance}
  \begin{sqt:definition}[\textbf{Definition:} Inheritance]
    A class (e.g. \verb|ClassA|) can inherit from another class (e.g. \verb|ClassB|).\\
    This means that \verb|ClassA| has all the members and methods of \verb|ClassB|,\\
    as well as any additional members and methods defined in \verb|ClassA|.\\
    Methods in \verb|ClassA| can also override methods in \verb|ClassB|.
  \end{sqt:definition}

  Suppose that class \verb|Base| exists.
  To define a class \verb|Derived| that inherits from \verb|Base|:
  \begin{itemize}
    \item The first member variable of \verb|Derived| must be an instance of \verb|Base|
    \item The constructor of \verb|Derived| must first call the constructor of \verb|Base|
    \item The destructor of \verb|Derived| must lastly call the destructor of \verb|Base|
  \end{itemize}
\end{frame}

\section{Generic Programming}

% Intro to templates
\begin{frame}{C++ Templates}
  \begin{sqt:definition}[\textbf{Definition:} Generic Programming]
    Writing algorithms and data structures \textbf{once}, usable with \textbf{any type}.
  \end{sqt:definition}

  Generic programming promotes:
  \begin{itemize}
    \item \textbf{Reusability} -- no need to rewrite code multiple times
    \item \textbf{Type safety} -- errors caught at compile time
    \item \textbf{Efficiency} -- no runtime overhead
  \end{itemize}

  In C++, this is achieved with \textbf{templates}.
\end{frame}


\begin{frame}[fragile]{C++ Templates}
  \begin{minted}{cpp}
    template <typename T>
    // ^ declares T as a placeholder type
  \end{minted}

  \texttt{typename} and \texttt{class} are interchangeable in template declarations.

  \vspace{0.4cm}
  Templates can be used to define:
  \begin{itemize}
    \item \textbf{Function templates}: generic functions over any type
    \item \textbf{Class templates}: generic data structures over any type
    \item \textbf{Variable templates}: generic constants over any type (C++14)
  \end{itemize}
\end{frame}

% Function templates
\begin{frame}[fragile]{Example: \texttt{max()} without function templates}
  \begin{minted}{cpp}
    int max(int x, int y) {
        return (x > y) ? x : y;
    }
    double max(double x, double y) {
        return (x > y) ? x : y;
    }
    char max(char x, char y) {
        return (x > y) ? x : y;
    }
  \end{minted}

  Same logic, repeated for every type. Can we do better?
\end{frame}


\begin{frame}[fragile]{Example: \texttt{max()} with function templates}
  \begin{minted}{cpp}
    #include <iostream>

    // "Template parameter declaration"
    template <typename T> 
    T max(T x, T y) {
        return (x > y) ? x : y;
    }

    int main(){
      std::cout << max(1, 2) << '\n'; // "2"
      std::cout << max(3.5, 7.5) << '\n'; // "7.5"
      std::cout << max('a', 'b') << '\n';  // "b"
      return 0;
    }

  \end{minted}

  \texttt{T} is a naming convention for a template type parameter (as in "type"). \\
  What about \texttt{max(1, 2.1)}?
\end{frame}

\begin{frame}[fragile]{Example: Multiple type parameters}
  \begin{minted}{cpp}
    #include <iostream>

    template <typename T, typename U> // additional types follow the same pattern
    auto max(T x, U y) {
        return (x > y) ? x : y;
    }

    int main(){
      std::cout << max(1, 2.1) << '\n'; // "2.1"
      return 0;
    }

  \end{minted}

  Using the \texttt{auto} keyword, the compiler will deduce what the return type should be.
\end{frame}

\begin{frame}[fragile]{Where have we seen function templates?}
  \texttt{std::sort} from \texttt{<algorithm>} works on any container of any type:
  \begin{minted}{cpp}
    std::vector<int> ints = {3, 1, 2};
    std::vector<double> doubles = {3.5, 1.1, 2.2};
    std::vector<std::string> strings = {"banana", "apple", "cherry"};

    std::sort(ints.begin(), ints.end());
    std::sort(doubles.begin(), doubles.end());
    std::sort(strings.begin(), strings.end());
  \end{minted}
  Under the hood, \texttt{std::sort} is a function template -- the same code,
  compiled for each type it is called with.
\end{frame}

% Class templates
\begin{frame}[fragile]{Example: Class templates}
  \begin{minted}{cpp}
    #include <iostream>
    template <typename T>
    class Pair {
      public:
        T x_;
        T y_;
        Pair(T x, T y) : x_(x), y_(y) {}
        void print() const {
            std::cout << x_ << " " << y_ << '\n';
        }
    };
    int main() {
        Pair<int> ints(10, 20);
        Pair<double> doubles(3.14, 6.28);
        ints.print();    // 10 20
        doubles.print(); // 3.14 6.28
        return 0;
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Where have we seen class templates?}
  \texttt{std::vector} from \texttt{<vector>} works with any type:
  \begin{minted}{cpp}
    std::vector<int> ints = {1, 2, 3};
    std::vector<double> doubles = {1.1, 2.2, 3.3};
    std::vector<std::string> strings = {"a", "b", "c"};
  \end{minted}

  \texttt{std::vector} is a class template -- the same data structure,
  compiled for each type it is instantiated with. \vspace{4pt}

  In fact, all standard library containers (e.g. \texttt{queue}, \texttt{stack}, \texttt{map}) are class templates.
\end{frame}
