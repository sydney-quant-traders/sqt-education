\begin{frame}[fragile]{Inheritance Revision}
  \begin{sqt:result}
    When \texttt{Derived} inherits from \texttt{Base}, each instance of 
    \texttt{Derived} contains an instance of \texttt{Base} embedded inside it.
  \end{sqt:result}
  \begin{minted}{cpp}
    struct Base { int a; char *b; };
    struct Derived : public Base { int c; Widget d; };

\begin{frame}[fragile]{Non-virtual methods}
  \begin{sqt:definition}
    A member function of \texttt{Widget}
    is a function that implicitly receives \texttt{Widget *this} as its first argument,
    and whose body is allowed to make reference to members variables of \texttt{Widget}.
  \end{sqt:definition}

  \begin{minted}{cpp}
    struct Base {
        int base_info_;
        void set(int x) { this->base_info_ = x; }
    };

    Base b = new Base{};
    b->set(5); // Base::set(b, 5);
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Non-virtual methods}
  Non-virtual methods are inherited by derived classes so that they can be used. 
  Passing a derived instance into the base member function is valid, since the derived
  instance has an embedded base instance within it.
  \begin{minted}{cpp}
    struct Base {
        int base_info_;
        void set(int x) { this->base_info_ = x; }
        void get(int x) { this->base_info_ = x; }
    };
    struct Derived : public Base { 
        int derived_info_; 
        void set(int x) { this->base_info_ = 10 * x; this->derived_info_ = x; }
    };

    Derived d = new Derived{};
    d->set(5);               // Derived::set(d, 5);
    std::cout << d->get(5);  // Base::get(static_cast<Base *>(d));
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Virtual methods}
  At runtime, a \code{Base *} may point to a \code{Derived1} or \code{Derived2} 
  instance and so on. \\
  This \emph{runtime type} may depend on runtime conditions (unknowable by compiler).\\
  \begin{minted}{cpp}
    struct Base {
        virtual void foo() {}
        void bar () {}
    };
    struct Derived1 : public Base {
        virtual void foo() override {}
        void bar () {}
    };
    /* define Derived2 similarly */
    Base *b = get_mystery_object();
    b->foo();         // virtual dispatch:  MYSTERY::foo(static_cast<MYSTERY*>(b));
    b->bar();         // static dispatch:   Base::bar(b);
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Vtable Dispatch}
  These terms are interchangeable in C++:
  \begin{itemize}
    \item subtype polymorphism
    \item dynamic/runtime polymorphism
    \item dynamic/runtime dispatch
  \end{itemize}

  This is when 
  \begin{itemize}
    \item you refer to a \emph{derived class instance} via a \emph{base class pointer}
    \item invoking a method through this pointer should invoke the \emph{derived class's methods}
  \end{itemize}

  \begin{sqt:result}
    Virtual table (vtable) dispatch is how C++ implements subtype polymorphism.
  \end{sqt:result}
\end{frame}

\begin{frame}[fragile]{Vtable Dispatch}
  The base class \code{processor} has a virtual method \code{process}.\\
  The function \code{process\_vector} runs the processor against a vector.
  \begin{minted}{cpp}
    /* 2.00 */
    struct processor { virtual void process(int input) {} };

    void process_vector(const std::vector<int>& nums, processor *p) {
        for (auto e : nums) {
            // dynamic dispatch: MYSTERY::process(static_cast<MYSTERY*>(p), e);
            p->process(e); 
        }
    }
  \end{minted}
  The derived classes are \code{printer} and \code{inserter}.
  \begin{itemize}
    \item \code{printer} processes an int by printing it.
    \item \code{inserter} processes an int by inserting it twice into a dst vector.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Vtable Implementation}
  We can emulate the behavior of virtual dispatch using function pointers. \\
  This is just for the sake of understanding how it works.

  \begin{minted}{cpp}
    /* 2.01 */
    struct processor_vtable_t {
        void(* process)(processor *, int);
    };

    struct processor {
        processor_vtable_t *vptr_; // may point to one of three vtables
        /* ... */
        void _virtual_process(int input) {
            auto process_fn_ptr = vptr_->process;
            process_fn_ptr(this, input);
        }
    };
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Vtable Implementation}
  \begin{minted}{cpp}
    /* 2.01 */
    // --- Global vtable instances ---
    processor_vtable_t _processor_vtable { .process = processor::_process };
    processor_vtable_t _printer_vtable { .process = printer::_process };
    processor_vtable_t _inserter_vtable { .process = inserter::_process };

    // --- Printer definition ---
    struct printer {
        processor base_{&_printer_vtable};

        // process() virtual method override and its static wrapper
        void process(int input) { std::cout << input << " "; }
        static void _process(processor *this_base, int input) {/* */}
    };
  \end{minted}
\end{frame}
