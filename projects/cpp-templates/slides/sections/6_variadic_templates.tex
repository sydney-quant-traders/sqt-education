\begin{frame}[fragile]{Analysis of \texttt{bounded\_vector::push\_back()}}
  \begin{minted}{cpp}
    template <typename F>
    void bounded_vector::push_back(F&& obj) {
        assert(size_ < Capacity);
        new (data_ + size_) T(std::forward<F>(obj));
        size_++;
    }
    /* ... */
    int main() {
        bounded_vector<intbox, 8> b;
        b.push_back(intbox{3});
    }
  \end{minted}

  \begin{enumerate}
    \item \texttt{intbox\{3\}} is \textbf{constructed} as a temporary in the caller's stack frame.
    \item Inside \verb|push_back|, it is \textbf{move constructed} into the vector's storage.
    \item The temporary is destroyed after the call to \verb|push_back()|.
  \end{enumerate}
\end{frame}

\begin{frame}[fragile]{Type Packs}
  \begin{sqt:definition}
    A \emph{template parameter pack} (type pack) contains \emph{multiple} types.
  \end{sqt:definition}

  \begin{sqt:result}
    Packs can be expanded by placing \verb|...| after the name of the pack.
    During template instantiation, the pack expands to the contained types, separated by commas.
  \end{sqt:result}

  \begin{minted}{cpp}
    template <typename... TypePack>
    void foo() {
        using expansion = std::tuple<TypePack...>;
    }

    int main() {
        foo<int>();
        foo<char, int, int*>();
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Type Patterns}
  \begin{sqt:definition}
    A \emph{pattern} is a type expression involving a type pack.
    Applying \texttt{...} to a pattern \emph{expands} it once for each element of the pack.
  \end{sqt:definition}

  Instantiate this template in cppinsights.io to understand how patterns are expanded.
  \begin{minted}{cpp}
    template <typename... TypePack>
    void foo() {
        using expansion = std::tuple<std::pair<int, TypePack*>...>;
        expansion x;
    }
  \end{minted}

  In this example, \textcolor{blue}{\texttt{std::pair<int, TypePack*>}} is the pattern which expands to
  \begin{minted}{cpp}
    std::pair<int, Type1*>, std::pair<int, Type2*>, std::pair<int, Type3*>
  \end{minted}
\end{frame}

\begin{frame}[fragile]{Object Packs}
  \begin{sqt:definition}
    An \emph{function parameter pack} (object pack) is a function parameter that accepts
    zero or more arguments. It is created by expanding a type pack pattern
    in a function's parameter list.
  \end{sqt:definition}

  \begin{minted}{cpp}
    template <typename... TypePack>
    void foo(TypePack... obj_pack) {
        std::tuple<TypePack...> x{obj_pack...};
        std::tuple<TypePack*...> y{(&obj_pack)...};
    }

    int main() {
        foo(5, 'q', "t");
    }
  \end{minted}
\end{frame}

\begin{frame}[fragile]{\texttt{emplace\_back()} with Variadic Templates}
  \begin{minted}{cpp}
    template <typename... TypePack>
    void emplace_back(TypePack&&... obj_pack) {
        assert(size_ < Capacity);
        new (data_ + size_) T(std::forward<TypePack>(obj_pack)...);
        size_++;
    }
  \end{minted}

  This single function definition handles:
  \begin{itemize}
    \item any number of constructor arguments (due to use of variadics)
    \item any combination of types (use of templates)
    \item any combination of value categories (use of forwarding reference)
  \end{itemize}
  And perfectly forwards these arguments through to the actual constructor\\ 
  (use of \texttt{std::forward}).
\end{frame}

\begin{frame}[fragile]{Unique Pointer}
  \verb|std::unique_ptr<T>| is a template class defined in \texttt{<memory>}.\\
  It contains a single \texttt{T*} raw pointer, which is passed in during construction.\\
  Methods \texttt{get()}, \texttt{release()}, \texttt{reset()} are ways to managing this pointer.\\
  The destructor calls \texttt{delete} on the raw pointer.\\
  Copy construction and assignment is disabled, trying to enforce \emph{unique ownership}.
  
  \begin{minted}{cpp}
    int main() {
        // construct a smart pointer that takes ownership of [42]
        int *raw_ptr1 = new int{42}; 
        std::unique_ptr<int> smart_ptr1{raw_ptr};
        // returns the pointer to [42], const operation
        auto p = smart_ptr1.get();
        // returns the pointer to [42], smart_ptr1 now points to nullptr
        p = smart_ptr1.release();
        // give smart_ptr ownership of [42] again
        smart_ptr1.reset(p);
        // end of scope: smart_ptr1 destructor runs, deleting [42]
    }
  \end{minted}
\end{frame}
